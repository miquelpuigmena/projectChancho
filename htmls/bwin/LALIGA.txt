        function decToFrac(dec) {
            dec = "" + dec;
            if (dec == '2.00' || dec == '2.0' || dec == '2') return 'Evens';
            if (dec.search('/') == -1) {
                dec = dec.split('.');
                if (dec.length == 1)
                    dec.push('00');

                dec[0] = parseInt(dec[0]) - 1;
                dec[1] = rtrim2(dec[1], '0');

                var num = parseInt(ltrim2(dec[0]+dec[1], '0'));
                var den = Math.pow(10, dec[1].length);
                var gcd = euclidean_gcd2(num, den);

                frac = num/gcd+'/'+den/gcd;
